class Solution:
    def constructArray(self, n, k):
        """
        :type n: int
        :type k: int
        :rtype: List[int]
        """
        list = []
        tmp = k
        start = 1
        for i in range(k + 1):
            if i % 2 == 0:
                list.append(start)
                start = start + 1
            else:
                list.append(tmp + 1)
                tmp = tmp - 1

        j = k + 1
        while j < n:
            list.append(j + 1)
            j = j + 1
        return list
